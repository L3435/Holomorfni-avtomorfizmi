int ITER = 1000;
real STEP = 0.3;
pair a = (0.6,-0.3);

pair aut(pair X){
  return X;
}

path Re(real y, pair A(pair)){
  guide g;
  for(int i = 0; i < 1000; ++i){
    g = g--(A((2*sqrt(1-y*y)*i/ITER - sqrt(1-y*y),y)));
  }
  path p = g;
  return p;
}

\begin{asy}
int ITER = 1000;
real STEP = 0.3;

  return exp(I*theta) * (X-a) / (1-conj(a)*X);
}

  guide g;

  path p = g;
  return p;
}
